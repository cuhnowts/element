```rust
#[tauri::command]
pub async fn validate_cli_tool(command: String) -> Result<bool, String> {
    use tokio::process::Command as TokioCommand;

    let result = tokio::time::timeout(
        std::time::Duration::from_secs(5),
        TokioCommand::new(&command)
            .arg("--version")
            .stdout(std::process::Stdio::null())
            .stderr(std::process::Stdio::null())
            .status()
    ).await;

    match result {
        Ok(Ok(status)) => Ok(status.success()),
        Ok(Err(_)) => Ok(false),  // Command not found or failed to spawn
        Err(_) => Ok(false),       // Timeout
    }
}
```

Note: Returns `Ok(false)` not `Err(...)` for missing tools — "not found" is a valid validation result, not an error.
