```typescript
import ReactMarkdown from "react-markdown";
import remarkGfm from "remark-gfm";
import rehypeHighlight from "rehype-highlight";

// CRITICAL: Memoize plugin arrays outside component to prevent
// react-markdown from reinitializing plugins on each render (Pitfall 1)
const remarkPlugins = [remarkGfm];
const rehypePlugins = [rehypeHighlight];

interface MarkdownRendererProps {
  content: string;
}

export function MarkdownRenderer({ content }: MarkdownRendererProps) {
  return (
    <ReactMarkdown
      remarkPlugins={remarkPlugins}
      rehypePlugins={rehypePlugins}
      components={{
        // Open links in external browser via target="_blank"
        a: ({ href, children, ...props }) => (
          <a href={href} target="_blank" rel="noopener noreferrer" {...props}>
            {children}
          </a>
        ),
      }}
    >
      {content}
    </ReactMarkdown>
  );
}
```

**Send message flow:**
```typescript
async function handleSend(content: string) {
  addUserMessage(content);
  startStreaming();

  // Build messages payload for Rust command
  const currentMessages = useHubChatStore.getState().messages;
  const payload: ChatMessagePayload[] = currentMessages.map(m => ({
    role: m.role,
    content: m.content,
  }));

  // System prompt: inject context manifest if available
  // For now, use a base system prompt. Phase 23 will provide the manifest.
  const systemPrompt = "You are a helpful project assistant. You help users manage their tasks, projects, and priorities. Be concise and actionable.";

  try {
    await hubChatSend(payload, systemPrompt);
  } catch (err) {
    // Error is also emitted via hub-chat-stream-error event
    // which the hook handles. This catch is for command-level errors.
    useHubChatStore.getState().setError(
      err instanceof Error ? err.message : String(err)
    );
  }
}
```

**Stop flow:**
```typescript
async function handleStop() {

}
```

**Layout structure:**
```tsx
<div className="flex flex-col h-full">
  {/* Separator between briefing and chat */}
  <Separator />

  {/* Message area - takes remaining space */}
  <ScrollArea ref={scrollRef} className="flex-1 px-8 py-4">
    {/* Suggestion chips shown when no messages */}
    {messages.length === 0 && !isStreaming && (
      <SuggestionChips onSelect={handleSend} />
    )}

    {/* Message list */}
    {messages.map((msg, i) => (
      <div key={msg.id} className={/* 16px gap between different senders, 8px same sender */}>
        <ChatMessage message={msg} />
      </div>
    ))}

    {/* Currently streaming message */}
    {isStreaming && streamingContent && (
      <div className="mt-4">
        <ChatMessage
          message={{ id: "streaming", role: "assistant", content: "", timestamp: Date.now() }}
          isCurrentlyStreaming
          streamingContent={streamingContent}
        />
      </div>
    )}

    {/* Error display */}
    {error && (
      <div className="text-sm mt-2" style={{ color: "var(--color-destructive)" }}>
        {error}
        <button onClick={() => /* resend last message */}>Retry</button>
      </div>
    )}

    {/* Scroll-to-bottom pill */}
    {showScrollButton && (
      <button onClick={scrollToBottom} aria-label="Scroll to latest"
        className="fixed ... rounded-full p-2 bg-card border">
        <ChevronDown size={16} />
      </button>
    )}
  </ScrollArea>

  {/* Input area - pinned to bottom */}
  <div className="px-8 py-3 border-t" style={{ borderColor: "var(--color-border)" }}>
    <ChatInput
      onSend={handleSend}
      isStreaming={isStreaming}
      onStop={handleStop}
    />
  </div>
</div>
```
