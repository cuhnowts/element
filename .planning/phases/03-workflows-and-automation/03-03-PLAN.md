```rust
use std::sync::Arc;
use tokio_cron_scheduler::{JobScheduler, Job};
use tauri::AppHandle;
use crate::db::connection::Database;
use crate::engine::executor::PipelineExecutor;
```

Implement:

- `pub async fn init_scheduler(app: AppHandle, db: Arc<std::sync::Mutex<Database>>) -> Result<JobScheduler, Box<dyn std::error::Error>>`:
  1. Create `JobScheduler::new().await?`
  2. Lock DB briefly, call db.list_active_schedules(), unlock
  3. For each active schedule:
     a. Check for missed runs: if schedule.last_run_at is Some, parse it as chrono DateTime, compare with now. If the cron expression indicates a run should have occurred between last_run_at and now, spawn a catch-up execution:
        ```rust
        let app_clone = app.clone();
        let db_clone = db.clone();
        let wf_id = schedule.workflow_id.clone();
        tokio::spawn(async move {
            catch_up_run(&app_clone, &db_clone, &wf_id).await;
        });
        ```
     b. Register recurring job using the cron expression:
        ```rust
        let cron_expr = schedule.cron_expression.clone();
        let app_clone = app.clone();
        let db_clone = db.clone();
        let wf_id = schedule.workflow_id.clone();
        let sched_id = schedule.id.clone();

        sched.add(Job::new_async(cron_expr.as_str(), move |_uuid, _lock| {
            let app = app_clone.clone();
            let db = db_clone.clone();
            let id = wf_id.clone();
            let sid = sched_id.clone();
            Box::pin(async move {
                scheduled_run(&app, &db, &id, &sid).await;
            })
        })?).await?;
        ```
  4. Start scheduler: `sched.start().await?`
  5. Return scheduler

- `async fn catch_up_run(app: &AppHandle, db: &Arc<std::sync::Mutex<Database>>, workflow_id: &str)`:
  1. Lock DB, get workflow, unlock
  2. Create PipelineExecutor::new(), execute with trigger_type "catch-up"

- `async fn scheduled_run(app: &AppHandle, db: &Arc<std::sync::Mutex<Database>>, workflow_id: &str, schedule_id: &str)`:
  1. Lock DB, get workflow, unlock
  2. Create PipelineExecutor::new(), execute with trigger_type "scheduled"
  3. Lock DB, update_schedule_last_run with current timestamp, unlock

- `pub async fn add_schedule_job(sched: &JobScheduler, app: AppHandle, db: Arc<std::sync::Mutex<Database>>, workflow_id: String, schedule_id: String, cron_expression: String) -> Result<uuid::Uuid, Box<dyn std::error::Error>>`:
  Register a single new job (called when user creates/updates a schedule). Returns the tokio-cron-scheduler job UUID.

- `pub fn compute_next_runs(cron_expression: &str, count: usize) -> Result<Vec<String>, String>`:
  Parse the cron expression using the cron crate (available via tokio-cron-scheduler dependency). Compute the next N (typically 3) run times from now. Return as Vec of ISO 8601 strings. This is called by the frontend (via a Tauri command) to display the CronPreview.

Implement all actions using the command wrappers. Key patterns:
- `fetchWorkflows`: call commands.listWorkflows(), set(workflows)
- `selectWorkflow(id)`: set selectedWorkflowId, call commands.getWorkflow(id), set selectedWorkflow, also fetchSchedule(id) and fetchRuns(id)
- `runWorkflow(id)`: set isRunning=true, stepStatuses={}, call commands.runWorkflow(id)
- `setStepRunning(progress)`: set currentStepIndex=progress.stepIndex, update stepStatuses[progress.stepIndex]="running"
- `setStepCompleted(progress)`: update stepStatuses[progress.stepIndex]="completed"
- `setStepFailed(progress)`: set isRunning=false, update stepStatuses[progress.stepIndex]="failed"

**After the store definition, set up event listeners** (module-level, runs once on import):
```typescript
// Set up Tauri event listeners for execution progress
listen<StepProgress>("workflow-step-started", (event) => {
  useWorkflowStore.getState().setStepRunning(event.payload);
});
listen<StepProgress>("workflow-step-completed", (event) => {
  useWorkflowStore.getState().setStepCompleted(event.payload);
});
listen<StepProgress>("workflow-step-failed", (event) => {
  useWorkflowStore.getState().setStepFailed(event.payload);
});
// Listen for workflow completion to refresh runs
listen("workflow-run-completed", () => {
  const state = useWorkflowStore.getState();
  if (state.selectedWorkflowId) {
    state.fetchRuns(state.selectedWorkflowId);
  }
  useWorkflowStore.setState({ isRunning: false, currentStepIndex: null });
});
```
