- Per D-02: Remember last center tab per project
- Per D-03: Remember drawer open/closed and active drawer tab per project
- Per D-14: `lastViewedAt` timestamps are in-memory only (NOT persisted)

```typescript
import { create } from "zustand";
import { persist } from "zustand/middleware";

interface ProjectWorkspaceState {
  centerTab: "detail" | "files";
  drawerOpen: boolean;
  drawerTab: "logs" | "history" | "terminal";
}

const DEFAULT_PROJECT_STATE: ProjectWorkspaceState = {
  centerTab: "detail",
  drawerOpen: true,
  drawerTab: "logs",
};

const IDLE_THRESHOLD_MS = 30 * 60 * 1000; // 30 minutes

interface WorkspaceState {
  // Existing global state
  drawerHeight: number;
  drawerOpen: boolean;
  calendarVisible: boolean;
  selectedTaskId: string | null;

  // Per-project state (persisted minus timestamps)
  projectStates: Record<string, ProjectWorkspaceState>;

  // Per-project timestamps (in-memory only, NOT persisted, per D-14)
  lastViewedTimestamps: Record<string, number>;

  // Existing actions
  setDrawerHeight: (height: number) => void;
  toggleDrawer: () => void;
  toggleCalendar: () => void;
  selectTask: (id: string | null) => void;

  // Per-project actions
  getProjectState: (projectId: string) => ProjectWorkspaceState;
  setProjectCenterTab: (projectId: string, tab: "detail" | "files") => void;
  setProjectDrawer: (projectId: string, open: boolean, tab?: "logs" | "history" | "terminal") => void;
  recordProjectView: (projectId: string) => void;
  shouldShowSummary: (projectId: string) => boolean;
}

export const useWorkspaceStore = create<WorkspaceState>()(
  persist(
    (set, get) => ({
      // Existing global state
      drawerHeight: 30,
      drawerOpen: true,
      calendarVisible: true,
      selectedTaskId: null,

      // Per-project state
      projectStates: {},
      lastViewedTimestamps: {},

      // Existing actions
      setDrawerHeight: (height) => set({ drawerHeight: height }),
      toggleDrawer: () => set((s) => ({ drawerOpen: !s.drawerOpen })),
      toggleCalendar: () => set((s) => ({ calendarVisible: !s.calendarVisible })),
      selectTask: (id) => set({ selectedTaskId: id }),

      // Per-project actions
      getProjectState: (projectId) => {
        return get().projectStates[projectId] ?? DEFAULT_PROJECT_STATE;
      },

      setProjectCenterTab: (projectId, tab) => {
        set((s) => ({
          projectStates: {
            ...s.projectStates,
            [projectId]: {
              ...(s.projectStates[projectId] ?? DEFAULT_PROJECT_STATE),
              centerTab: tab,
            },
          },
        }));
      },

      setProjectDrawer: (projectId, open, tab) => {
        set((s) => {
          const current = s.projectStates[projectId] ?? DEFAULT_PROJECT_STATE;
          return {
            projectStates: {
              ...s.projectStates,
              [projectId]: {
                ...current,
                drawerOpen: open,
                drawerTab: tab ?? current.drawerTab,
              },
            },
          };
        });
      },

      recordProjectView: (projectId) => {
        set((s) => ({
          lastViewedTimestamps: {
            ...s.lastViewedTimestamps,
            [projectId]: Date.now(),
          },
        }));
      },

      shouldShowSummary: (projectId) => {
        const lastViewed = get().lastViewedTimestamps[projectId];
        // First view in session (no timestamp) -> always show (per D-04)
        if (!lastViewed) return true;
        // Show if 30+ minutes idle
        return Date.now() - lastViewed >= IDLE_THRESHOLD_MS;
      },
    }),
    {
      name: "element-workspace",
      partialize: (state) => ({
        drawerHeight: state.drawerHeight,
        drawerOpen: state.drawerOpen,
        calendarVisible: state.calendarVisible,
        projectStates: state.projectStates,
        // Do NOT persist selectedTaskId (task may not exist on next launch)
        // Do NOT persist lastViewedTimestamps (must reset on restart per D-14)
      }),
    }
  )
);
```

**1. Create `src/stores/contextSlice.ts` (new file):**

This slice manages the AI context summary and suggestion lifecycle. Key design:
- Uses `currentRequestProjectId` to prevent race conditions on fast project switching (per pitfall #2 in research)
- Summary and suggestions are per-project, cleared on project switch
- Dismissed suggestion IDs tracked per-project to prevent reappearance (per D-10)
- On-demand mode gating is done at the component level, not in the store

```typescript
import type { StateCreator } from "zustand";
import type { AppStore } from "./index";
import type { ContextSummary, AiSuggestion } from "../types/ai";
import { api } from "../lib/tauri";

export interface ContextSlice {
  // State
  contextSummary: ContextSummary | null;
  contextSuggestions: AiSuggestion[];
  isGeneratingSummary: boolean;
  summaryError: string | null;
  currentRequestProjectId: string | null;
  dismissedSummaryProjectIds: Set<string>;
  dismissedSuggestionIds: Set<string>;

  // Actions
  generateContextSummary: (projectId: string) => Promise<void>;
  generateSuggestions: (projectId: string) => Promise<void>;
  dismissSummary: (projectId: string) => void;
  dismissSuggestion: (suggestionId: string) => void;
  clearContextState: () => void;
}

export const createContextSlice: StateCreator<
  AppStore,
  [],
  [],
  ContextSlice
> = (set, get) => ({
  contextSummary: null,
  contextSuggestions: [],
  isGeneratingSummary: false,
  summaryError: null,
  currentRequestProjectId: null,
  dismissedSummaryProjectIds: new Set(),
  dismissedSuggestionIds: new Set(),

  generateContextSummary: async (projectId: string) => {
    // Check if summary was already dismissed for this project (don't re-show)
    if (get().dismissedSummaryProjectIds.has(projectId)) return;

    set({
      isGeneratingSummary: true,
      summaryError: null,
      contextSummary: null,
      currentRequestProjectId: projectId,
    });

    try {
      const summary = await api.generateContextSummary(projectId);
      // Race condition guard: only apply if still on same project (per pitfall #2)
      if (get().currentRequestProjectId === projectId) {
        set({ contextSummary: summary, isGeneratingSummary: false });
      }
    } catch (e) {
      if (get().currentRequestProjectId === projectId) {
        set({
          summaryError: String(e),
          isGeneratingSummary: false,
        });
      }
    }
  },

  generateSuggestions: async (projectId: string) => {
    try {
      const suggestions = await api.generateSuggestions(projectId);
      // Race condition guard
      if (get().currentRequestProjectId === projectId) {
        // Filter out any previously dismissed suggestions
        const dismissed = get().dismissedSuggestionIds;
        const filtered = suggestions.filter((s) => !dismissed.has(s.id));
        set({ contextSuggestions: filtered });
      }
    } catch {
      // Suggestions are non-critical; silently fail
      if (get().currentRequestProjectId === projectId) {
        set({ contextSuggestions: [] });
      }
    }
  },

  dismissSummary: (projectId: string) => {
    set((s) => {
      const newDismissed = new Set(s.dismissedSummaryProjectIds);
      newDismissed.add(projectId);
      return {
        contextSummary: null,
        dismissedSummaryProjectIds: newDismissed,
      };
    });
  },

  dismissSuggestion: (suggestionId: string) => {
    set((s) => {
      const newDismissed = new Set(s.dismissedSuggestionIds);
      newDismissed.add(suggestionId);
      return {
        contextSuggestions: s.contextSuggestions.filter(
          (sg) => sg.id !== suggestionId
        ),
        dismissedSuggestionIds: newDismissed,
      };
    });
  },

  clearContextState: () => {
    set({
      contextSummary: null,
      contextSuggestions: [],
      isGeneratingSummary: false,
      summaryError: null,
      currentRequestProjectId: null,
      dismissedSummaryProjectIds: new Set(),
      dismissedSuggestionIds: new Set(),
    });
  },
});
```
