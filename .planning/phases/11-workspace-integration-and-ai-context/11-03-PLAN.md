**1. Create `src/components/center/ContextSummaryCard.tsx`:**

This component handles three states: loading, error, and content (both AI-generated and fallback template).

Per UI-SPEC:
- Container: `rounded-lg bg-card p-4` with `border-l-2` using `--color-chart-2` (oklch(0.6 0.118 184.714))
- Header: Bot icon (16px, muted) + "Where was I?" (16px semibold) + dismiss button
- Fallback variant: BarChart3 icon + "Project Status" header + no dismiss button
- Loading: 3 Skeleton lines (w-3/4, w-1/2, w-2/3) in card container with `aria-busy="true"`
- Error: destructive background with retry button and copy "Couldn't load your project summary. Check your AI provider in Settings and try again."
- Appear: `animate-in fade-in slide-in-from-top-2 duration-300`

```tsx
import { Bot, BarChart3, X, RefreshCw } from "lucide-react";
import { Button } from "@/components/ui/button";
import { Skeleton } from "@/components/ui/skeleton";
import type { ContextSummary } from "@/types/ai";

interface ContextSummaryCardProps {
  summary: ContextSummary | null;
  isLoading: boolean;
  error: string | null;
  onDismiss: () => void;
  onRetry: () => void;
}

export function ContextSummaryCard({
  summary,
  isLoading,
  error,
  onDismiss,
  onRetry,
}: ContextSummaryCardProps) {
  // Loading state
  if (isLoading) {
    return (
      <div
        className="rounded-lg bg-card p-4 border-l-2"
        style={{ borderColor: "oklch(0.6 0.118 184.714)" }}
        aria-busy="true"
        aria-label="Loading project summary"
      >
        <div className="space-y-2">
          <Skeleton className="h-4 w-3/4" />
          <Skeleton className="h-4 w-1/2" />
          <Skeleton className="h-4 w-2/3" />
        </div>
      </div>
    );
  }

  // Error state
  if (error) {
    return (
      <div className="rounded-lg bg-destructive/10 p-4">
        <p className="mb-2 text-sm text-destructive">
          Couldn't load your project summary. Check your AI provider in Settings and try again.
        </p>
        <Button variant="outline" size="sm" onClick={onRetry}>
          <RefreshCw className="size-3 mr-1" />
          Retry
        </Button>
      </div>
    );
  }

  if (!summary) return null;

  // Determine if this is a fallback (non-AI) summary
  const isFallback = !summary.isAiGenerated;
  const Icon = isFallback ? BarChart3 : Bot;
  const title = isFallback ? "Project Status" : "Where was I?";

  return (
    <div
      className="rounded-lg bg-card p-4 border-l-2 animate-in fade-in slide-in-from-top-2 duration-300"
      style={{ borderColor: "oklch(0.6 0.118 184.714)" }}
      role="status"
      aria-live="polite"
    >
      <div className="flex items-start justify-between mb-2">
        <div className="flex items-center gap-2">
          <Icon className="size-4 text-muted-foreground" aria-hidden="true" />
          <span className="text-base font-semibold">{title}</span>
        </div>
        {/* Fallback cards don't have dismiss button per UI-SPEC */}
        {!isFallback && (
          <Button
            variant="ghost"
            size="icon-xs"
            onClick={onDismiss}
            aria-label="Dismiss summary"
          >
            <X className="size-3" />
          </Button>
        )}
      </div>
      <ul className="space-y-1 pl-6">
        {summary.bullets.map((bullet, i) => (
          <li key={i} className="flex items-start gap-2 text-sm">
            <span className="text-muted-foreground mt-0.5">-</span>
            <span>{bullet}</span>
          </li>
        ))}
      </ul>
      {summary.note && (
        <p className="mt-2 text-xs text-muted-foreground italic">
          {summary.note}
        </p>
      )}
    </div>
  );
}
```

The full rewritten component:

```tsx
import { useEffect, useState, useRef, useCallback } from "react";
import { Input } from "@/components/ui/input";
import { Textarea } from "@/components/ui/textarea";
import { useStore } from "@/stores";
import { useWorkspaceStore } from "@/stores/useWorkspaceStore";
import { api } from "@/lib/tauri";
import { ContextSummaryCard } from "./ContextSummaryCard";
import { AiSuggestionCard } from "./AiSuggestionCard";

export function ProjectDetail() {
  const selectedProjectId = useStore((s) => s.selectedProjectId);
  const projects = useStore((s) => s.projects);
  const loadProjects = useStore((s) => s.loadProjects);
  const tasks = useStore((s) => s.tasks);

  // Context slice state
  const contextSummary = useStore((s) => s.contextSummary);
  const contextSuggestions = useStore((s) => s.contextSuggestions);
  const isGeneratingSummary = useStore((s) => s.isGeneratingSummary);
  const summaryError = useStore((s) => s.summaryError);
  const generateContextSummary = useStore((s) => s.generateContextSummary);
  const generateSuggestions = useStore((s) => s.generateSuggestions);
  const dismissSummary = useStore((s) => s.dismissSummary);
  const dismissSuggestion = useStore((s) => s.dismissSuggestion);

  // AI assist (for scaffold action)
  const requestAiAssist = useStore((s) => s.requestAiAssist);

  // Workspace store
  const shouldShowSummary = useWorkspaceStore((s) => s.shouldShowSummary);
  const recordProjectView = useWorkspaceStore((s) => s.recordProjectView);
  const selectTask = useWorkspaceStore((s) => s.selectTask);

  const project = projects.find((p) => p.id === selectedProjectId);

  const [name, setName] = useState("");
  const [description, setDescription] = useState("");
  const descTimer = useRef<ReturnType<typeof setTimeout> | null>(null);

  useEffect(() => {
    if (project) {
      setName(project.name);
      setDescription(project.description);
    }
  }, [project?.id]);

  // Project switch trigger: generate AI summary and suggestions
  useEffect(() => {
    if (!selectedProjectId) return;

    // Phase 10 integration point: read project.aiMode when available
    // For now, default to "track-suggest" (show AI features)
    // When Phase 10 lands, replace with: const aiMode = project?.aiMode ?? "track-suggest";
    const aiMode = "track-suggest";

    // D-11: On-demand mode shows no AI features
    if (aiMode === "on-demand") return;

    // D-04: Check idle threshold
    if (shouldShowSummary(selectedProjectId)) {
      generateContextSummary(selectedProjectId);
      generateSuggestions(selectedProjectId);
    }

    // Record this view (updates lastViewedTimestamp)
    recordProjectView(selectedProjectId);
  }, [selectedProjectId]);

  if (!project) return null;

  const handleNameBlur = async () => {
    if (name !== project.name && name.trim()) {
      await api.updateProject(project.id, name.trim(), description);
      loadProjects();
    }
  };

  const handleDescriptionChange = (value: string) => {
    setDescription(value);
    if (descTimer.current) clearTimeout(descTimer.current);
    descTimer.current = setTimeout(async () => {
      await api.updateProject(project.id, name, value);
      loadProjects();
    }, 800);
  };

  const handleRetry = useCallback(() => {
    if (selectedProjectId) {
      generateContextSummary(selectedProjectId);
      generateSuggestions(selectedProjectId);
    }
  }, [selectedProjectId, generateContextSummary, generateSuggestions]);

  const handleSuggestionAction = useCallback(
    (suggestion: { type: string; taskId: string }) => {
      if (suggestion.type === "go-to-task" && suggestion.taskId) {
        // Navigate to the task in the sidebar
        selectTask(suggestion.taskId);
      } else if (suggestion.type === "scaffold" && suggestion.taskId) {
        // Trigger AI scaffold for the task
        requestAiAssist(suggestion.taskId);
        selectTask(suggestion.taskId);
      }
    },
    [selectTask, requestAiAssist]
  );

  const taskCount = tasks.length;

  // Phase 10 integration point: read aiMode from project
  const aiMode = "track-suggest";
  const showAiFeatures = aiMode !== "on-demand";

  // Determine if summary card should show (not dismissed, and either loading/error/has content)
  const showSummaryCard =
    showAiFeatures &&
    (isGeneratingSummary || summaryError || contextSummary !== null);

  // Suggestion cards only in Track+Suggest or Track+Auto-execute modes
  const showSuggestions =
    showAiFeatures && contextSuggestions.length > 0;

  return (
    <div className="max-w-2xl mx-auto space-y-6">
      {/* AI Cards Section (per D-07: at top, above project content) */}
      {(showSummaryCard || showSuggestions) && (
        <div className="space-y-3 mb-6">
          {showSummaryCard && (
            <ContextSummaryCard
              summary={contextSummary}
              isLoading={isGeneratingSummary}
              error={summaryError}
              onDismiss={() => {
                if (selectedProjectId) dismissSummary(selectedProjectId);
              }}
              onRetry={handleRetry}
            />
          )}
          {showSuggestions &&
            contextSuggestions.map((suggestion, i) => (
              <AiSuggestionCard
                key={suggestion.id}
                suggestion={suggestion}
                index={i}
                onAction={() => handleSuggestionAction(suggestion)}
                onDismiss={() => dismissSuggestion(suggestion.id)}
              />
            ))}
        </div>
      )}

      <div>
        <span className="text-xs font-semibold tracking-wide uppercase text-muted-foreground block mb-1">
          Project
        </span>
        <Input
          value={name}
          onChange={(e) => setName(e.target.value)}
          onBlur={handleNameBlur}
          onKeyDown={(e) => {
            if (e.key === "Enter") e.currentTarget.blur();
          }}
          className="text-2xl font-semibold border-none shadow-none px-0 focus-visible:ring-0 bg-transparent"
          placeholder="Project name"
        />
      </div>

      <div>
        <span className="text-xs font-semibold tracking-wide uppercase text-muted-foreground block mb-2">
          Description
        </span>
        <Textarea
          value={description}
          onChange={(e) => handleDescriptionChange(e.target.value)}
          placeholder="What is this project about? Add goals, context, or notes..."
          className="min-h-[120px] resize-none"
        />
      </div>

      <div className="grid grid-cols-2 gap-4 text-sm">
        <div>
          <span className="text-xs font-semibold tracking-wide uppercase text-muted-foreground block mb-1">
            Tasks
          </span>
          <span>{taskCount}</span>
        </div>
        <div>
          <span className="text-xs font-semibold tracking-wide uppercase text-muted-foreground block mb-1">
            Created
          </span>
          <span>
            {new Date(project.createdAt).toLocaleDateString(undefined, {
              month: "short",
              day: "numeric",
              year: "numeric",
            })}
          </span>
        </div>
      </div>
    </div>
  );
}
```
