**2. Create `src-tauri/src/ai/context.rs` (new file):**

This module gathers project data for AI context. It uses raw SQL for aggregation rather than the existing `list_tasks` method to avoid deserializing every task field.

```rust
use crate::db::connection::Database;
use crate::ai::types::{ContextSummary, ProjectSuggestion};

pub struct ProjectContext {
    pub project_name: String,
    pub project_id: String,
    pub total_tasks: usize,
    pub total_complete: usize,
    pub recently_completed: usize,
    pub in_progress: usize,
    pub blocked: usize,
    pub next_incomplete_tasks: Vec<(String, String)>, // (id, title) pairs
}

pub fn gather_project_context(db: &Database, project_id: &str) -> Result<ProjectContext, String> {
    // Get project name
    let project = db.get_project(project_id).map_err(|e| format!("Project not found: {}", e))?;

    // Aggregate task counts using raw SQL (single query, no full deserialization)
    let mut stmt = db.conn().prepare(
        "SELECT
            COUNT(*) as total,
            SUM(CASE WHEN status = 'complete' THEN 1 ELSE 0 END) as complete,
            SUM(CASE WHEN status = 'in-progress' THEN 1 ELSE 0 END) as in_progress,
            SUM(CASE WHEN status = 'blocked' THEN 1 ELSE 0 END) as blocked,
            SUM(CASE WHEN status = 'complete' AND updated_at >= datetime('now', '-7 days') THEN 1 ELSE 0 END) as recent
         FROM tasks WHERE project_id = ?1"
    ).map_err(|e| e.to_string())?;

    let (total, complete, in_progress, blocked, recent) = stmt.query_row(
        rusqlite::params![project_id],
        |row| {
            Ok((
                row.get::<_, i64>(0)? as usize,
                row.get::<_, i64>(1)? as usize,
                row.get::<_, i64>(2)? as usize,
                row.get::<_, i64>(3)? as usize,
                row.get::<_, i64>(4)? as usize,
            ))
        }
    ).map_err(|e| e.to_string())?;

    // Get next 3 incomplete tasks (id + title)
    let mut next_stmt = db.conn().prepare(
        "SELECT id, title FROM tasks WHERE project_id = ?1 AND status IN ('pending', 'in-progress') ORDER BY CASE WHEN status = 'in-progress' THEN 0 ELSE 1 END, created_at ASC LIMIT 3"
    ).map_err(|e| e.to_string())?;

    let next_incomplete: Vec<(String, String)> = next_stmt.query_map(
        rusqlite::params![project_id],
        |row| Ok((row.get(0)?, row.get(1)?))
    ).map_err(|e| e.to_string())?
    .filter_map(|r| r.ok())
    .collect();

    Ok(ProjectContext {
        project_name: project.name,
        project_id: project_id.to_string(),
        total_tasks: total,
        total_complete: complete,
        recently_completed: recent,
        in_progress,
        blocked,
        next_incomplete_tasks: next_incomplete,
    })
}

pub fn build_template_summary(data: &ProjectContext) -> ContextSummary {
    let mut bullets = Vec::new();

    if data.recently_completed > 0 {
        bullets.push(format!("{} task(s) completed in the last 7 days", data.recently_completed));
    }
    if data.in_progress > 0 {
        bullets.push(format!("{} task(s) currently in progress", data.in_progress));
    }
    if data.blocked > 0 {
        bullets.push(format!("{} task(s) blocked -- may need attention", data.blocked));
    }
    bullets.push(format!(
        "Overall progress: {}/{} tasks complete",
        data.total_complete, data.total_tasks
    ));
    if !data.next_incomplete_tasks.is_empty() {
        let names: Vec<&str> = data.next_incomplete_tasks.iter().map(|(_, t)| t.as_str()).collect();
        bullets.push(format!("Next up: {}", names.join(", ")));
    }

    ContextSummary {
        project_id: data.project_id.clone(),
        bullets,
        is_ai_generated: false,
        note: Some("Configure an AI provider in Settings for smarter summaries.".into()),
        suggested_next: data.next_incomplete_tasks.iter().map(|(_, t)| t.clone()).collect(),
    }
}

pub fn build_template_suggestions(data: &ProjectContext) -> Vec<ProjectSuggestion> {
    let mut suggestions = Vec::new();

    // Suggest blocked tasks
    // (We only have count, not individual blocked tasks from aggregation -- skip for template)

    // Suggest next incomplete tasks as "go-to-task" suggestions
    for (id, title) in data.next_incomplete_tasks.iter().take(3) {
        suggestions.push(ProjectSuggestion {
            id: uuid::Uuid::new_v4().to_string(),
            project_id: data.project_id.clone(),
            suggestion_type: "go-to-task".to_string(),
            task_id: id.clone(),
            text: format!("Continue working on \"{}\"", title),
            action_label: "Go to task".to_string(),
        });
    }

    suggestions
}
```

use crate::ai::context::ProjectContext;

pub fn build_context_summary_request(context: &ProjectContext) -> CompletionRequest {
    let system = r#"You are a project assistant. Given project progress data, generate a concise "where was I?" summary. Respond with valid JSON only, no markdown wrapping.

JSON schema:
{
  "bullets": ["bullet 1", "bullet 2", ...],
  "suggested_next": ["action 1", "action 2"]
}

Rules:
- 3-5 bullet points maximum for bullets
- 1-3 suggested next actions
- Be concise and actionable
- Focus on what changed recently and what needs attention
- If tasks are blocked, mention them
- If progress is high, highlight how close to completion"#;

    let next_tasks: Vec<&str> = context.next_incomplete_tasks.iter().map(|(_, t)| t.as_str()).collect();

    let user = format!(
        "Project: {}\nTotal tasks: {} ({} complete, {} in-progress, {} blocked)\nRecently completed (7d): {}\nNext incomplete tasks: {}",
        context.project_name,
        context.total_tasks, context.total_complete, context.in_progress, context.blocked,
        context.recently_completed,
        if next_tasks.is_empty() { "(none)".to_string() } else { next_tasks.join(", ") }
    );

    CompletionRequest {
        system_prompt: system.to_string(),
        user_message: user,
        max_tokens: 512,
        temperature: 0.3,
    }
}

pub fn build_suggestions_request(context: &ProjectContext) -> CompletionRequest {
    let system = r#"You are a project assistant. Given project progress data, suggest actionable next steps. Respond with valid JSON only, no markdown wrapping.

JSON schema:
{
  "suggestions": [
    {
      "type": "go-to-task" | "scaffold",
      "task_id": "id of the task if referencing one, or empty string",
      "task_title": "title of the referenced task",
      "text": "Suggestion text explaining what to do",
      "action_label": "Go to task" | "Scaffold with AI"
    }
  ]
}

Rules:
- Maximum 3 suggestions
- Focus on: stalled tasks (in-progress too long), blocked items, nearly-complete phases, untouched tasks
- "go-to-task" = navigate to a specific task. "scaffold" = suggest AI scaffolding for a task
- Keep text under 100 characters
- Be specific -- reference actual task titles"#;

    let tasks_info: Vec<String> = context.next_incomplete_tasks.iter()
        .map(|(id, title)| format!("  - [{}] {}", id, title))
        .collect();

    let user = format!(
        "Project: {}\nTotal: {} tasks ({} complete, {} in-progress, {} blocked)\nRecently completed (7d): {}\nIncomplete tasks:\n{}",
        context.project_name,
        context.total_tasks, context.total_complete, context.in_progress, context.blocked,
        context.recently_completed,
        if tasks_info.is_empty() { "  (none)".to_string() } else { tasks_info.join("\n") }
    );

    CompletionRequest {
        system_prompt: system.to_string(),
        user_message: user,
        max_tokens: 512,
        temperature: 0.3,
    }
}

pub fn parse_context_summary_response(content: &str, project_id: &str) -> Result<ContextSummary, AiError> {
    // Try direct parse
    #[derive(Deserialize)]
    struct SummaryJson {
        bullets: Vec<String>,
        suggested_next: Option<Vec<String>>,
    }

    let parsed = try_parse_json::<SummaryJson>(content)
        .map_err(|_| AiError::Parse(format!("Failed to parse context summary JSON: {}", &content[..content.len().min(200)])))?;

    Ok(ContextSummary {
        project_id: project_id.to_string(),
        bullets: parsed.bullets,
        is_ai_generated: true,
        note: None,
        suggested_next: parsed.suggested_next.unwrap_or_default(),
    })
}

pub fn parse_suggestions_response(content: &str, project_id: &str) -> Result<Vec<ProjectSuggestion>, AiError> {
    #[derive(Deserialize)]
    struct SuggestionJson {
        #[serde(rename = "type")]
        suggestion_type: String,
        task_id: Option<String>,
        #[allow(dead_code)]
        task_title: Option<String>,
        text: String,
        action_label: Option<String>,
    }

    #[derive(Deserialize)]
    struct SuggestionsJson {
        suggestions: Vec<SuggestionJson>,
    }

    let parsed = try_parse_json::<SuggestionsJson>(content)
        .map_err(|_| AiError::Parse(format!("Failed to parse suggestions JSON: {}", &content[..content.len().min(200)])))?;

    Ok(parsed.suggestions.into_iter().map(|s| {
        let action_label = s.action_label.unwrap_or_else(|| {
            if s.suggestion_type == "scaffold" { "Scaffold with AI".to_string() } else { "Go to task".to_string() }
        });
        ProjectSuggestion {
            id: uuid::Uuid::new_v4().to_string(),
            project_id: project_id.to_string(),
            suggestion_type: s.suggestion_type,
            task_id: s.task_id.unwrap_or_default(),
            text: s.text,
            action_label,
        }
    }).collect())
}

/// Helper: try parsing JSON directly or from a markdown code block
fn try_parse_json<T: serde::de::DeserializeOwned>(content: &str) -> Result<T, String> {
    if let Ok(val) = serde_json::from_str::<T>(content) {
        return Ok(val);
    }
    let re = regex::Regex::new(r"```(?:json)?\s*\n?([\s\S]*?)\n?```")
        .map_err(|e| format!("Regex error: {}", e))?;
    if let Some(caps) = re.captures(content) {
        if let Some(json_str) = caps.get(1) {
            if let Ok(val) = serde_json::from_str::<T>(json_str.as_str()) {
                return Ok(val);
            }
        }
    }
    Err(format!("Could not parse JSON from content"))
}

use crate::ai::context;

#[tauri::command]
pub async fn generate_context_summary(
    project_id: String,
    db_state: State<'_, Arc<Mutex<Database>>>,
    gateway: State<'_, AiGateway>,
) -> Result<crate::ai::types::ContextSummary, String> {

    let context_data = {
        let db = db_state.lock().map_err(|e| e.to_string())?;
        context::gather_project_context(&db, &project_id)?
    };
    // Lock dropped here

    // 2. Try AI provider
    let provider_result = {
        let db = db_state.lock().map_err(|e| e.to_string())?;
        gateway.get_default_config(&db)
    };

    match provider_result {
        Ok(config) => {
            let provider = gateway.build_provider(&config).map_err(|e| e.to_string())?;
            let request = prompts::build_context_summary_request(&context_data);
            match provider.complete(request).await {
                Ok(response) => {

                Err(_) => {
                    // AI failed, fall back to template

                }
            }
        }
        Err(_) => {
            // No provider configured, use template fallback (per D-06)
            Ok(context::build_template_summary(&context_data))
        }
    }
}
