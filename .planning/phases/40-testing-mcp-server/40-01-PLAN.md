       ```typescript
       import { spawn } from "node:child_process";
       import type { SpawnResult } from "./types.js";

       export function runCommand(
         command: string,
         args: string[],
         cwd: string,
         timeoutMs: number = 120_000
       ): Promise<SpawnResult> {
         return new Promise((resolve, reject) => {
           const proc = spawn(command, args, {
             cwd,
             shell: false,
             stdio: ["ignore", "pipe", "pipe"],
             timeout: timeoutMs,
           });

           const chunks: Buffer[] = [];
           const errChunks: Buffer[] = [];

           proc.stdout.on("data", (d: Buffer) => chunks.push(d));
           proc.stderr.on("data", (d: Buffer) => errChunks.push(d));

           proc.on("close", (code) => {
             resolve({
               stdout: Buffer.concat(chunks).toString("utf-8"),
               stderr: Buffer.concat(errChunks).toString("utf-8"),
               exitCode: code ?? 1,
             });
           });

           proc.on("error", reject);
         });
       }
       ```

  <behavior>
    Vitest parser (parseVitestOutput):
    - Input: Vitest JSON reporter output string -> Output: TestResult[]
    - Passed test: status "passed", name from assertionResults.fullName, file from testResults.name, duration from assertionResults.duration
    - Failed test: status "failed", error contains failureMessages joined
    - Pending/todo tests: status "pending" or "todo"
    - Suite ancestry: suite from ancestorTitles joined with " > "

    Vitest parser (parseVitestList):
    - Input: vitest list --json output string -> Output: TestEntry[]
    - Each entry has runner="vitest", file, name, suite

    Cargo parser (parseCargoTestOutput):
    - Input: cargo test stdout string + stderr string -> Output: TestResult[]
    - "test mod::name ... ok" -> status "passed", module parsed from "::" path
    - "test mod::name ... FAILED" -> status "failed"
    - "test mod::name ... ignored" -> status "ignored"
    - Failure details extracted from "failures:" section stdout
    - Compilation failure (non-zero exit, no test lines) -> single TestResult with error from stderr

    Cargo parser (parseCargoTestList):
    - Input: cargo test -- --list stdout -> Output: TestEntry[]
    - Lines like "module::path::test_name: test" -> TestEntry with runner="cargo"

    Coverage parser (parseCoverageReport):
    - Input: Istanbul coverage JSON string -> Output: CoverageGap[]
    - Functions with f[key] === 0 are uncovered
    - Statements with s[key] === 0 counted
    - Files with zero uncovered functions omitted from output
    - Empty/invalid JSON -> throws descriptive error
  </behavior>
