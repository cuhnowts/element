   ```rust
       let handle = app.handle();
       let state: State<'_, Arc<Mutex<Database>>> = handle.state();
       let trigger: State<'_, ManifestRebuildTrigger> = handle.state();
       let result = create_task(
           handle.clone(),
           state,
           trigger,
           None, None, "Test Task".into(), None, None, None, None, None, None, None, None, None, None
       ).await;
       assert!(result.is_ok());
       assert_eq!(result.unwrap().title, "Test Task");
   }
   ```
