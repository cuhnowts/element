    import { Button } from "../ui/button";
    import { Label } from "../ui/label";

    interface ScopeInputFormProps {

    **Component 3: `src/components/center/OnboardingWaitingCard.tsx`** (per D-10)

    Waiting state card shown while CLI tool runs. Contains:
    - Card: `bg-card rounded-lg p-12 text-center max-w-md mx-auto`
    - Loader2 icon spinning (24px, `text-muted-foreground`, `animate-spin`) -- or CheckCircle2 if plan detected
    - Heading: "AI setup in progress..." (or "Plan ready for review" when detected)
    - Scope summary: truncated to 120 chars, `text-muted-foreground text-xs`
    - Hint: "Check the Terminal tab to interact with the AI." (`text-muted-foreground text-xs`)
    - "Stop AI setup" button (ghost variant, `text-destructive`)
    - Props: `{ scope: string; planDetected: boolean; onCancel: () => void; }`

    ```tsx
    import { Loader2, CheckCircle2 } from "lucide-react";
    import { Button } from "../ui/button";

    interface OnboardingWaitingCardProps {
      scope: string;
      planDetected: boolean;
      onCancel: () => void;
    }

    export function OnboardingWaitingCard({ scope, planDetected, onCancel }: OnboardingWaitingCardProps) {
      const truncatedScope = scope.length > 120 ? scope.slice(0, 120) + "..." : scope;
      return (
        <div className="bg-card rounded-lg p-12 text-center max-w-md mx-auto mt-12" role="status">
          {planDetected ? (
            <CheckCircle2 className="h-6 w-6 text-chart-2 mx-auto mb-4" />
          ) : (
            <Loader2 className="h-6 w-6 text-muted-foreground animate-spin mx-auto mb-4" aria-label="AI planning in progress" />
          )}
          <h3 className="text-base font-semibold mb-3">
            {planDetected ? "Plan ready for review" : "AI setup in progress..."}
          </h3>
          <p className="text-xs text-muted-foreground mb-2">
            Scope: &ldquo;{truncatedScope}&rdquo;
          </p>
          {!planDetected && (
            <>
              <p className="text-xs text-muted-foreground mb-6">
                Check the Terminal tab to interact with the AI.
              </p>
              <Button variant="ghost" className="text-destructive" onClick={onCancel}>
                Stop AI setup
              </Button>
            </>
          )}
        </div>
      );
    }
    ```
