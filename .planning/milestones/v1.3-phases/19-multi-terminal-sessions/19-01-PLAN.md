    import { create } from "zustand";

    export interface TerminalSession {
      id: string;                    // crypto.randomUUID()
      projectId: string;
      name: string;                  // "Phase 19: Multi-Terminal", "Dev Server", "Shell 1"
      type: "ai" | "shell";

    Store interface per RESEARCH.md Pattern 1:
    - `sessions: Record<string, TerminalSession[]>` — keyed by projectId
    - `activeSessionId: Record<string, string | null>` — keyed by projectId
    - `createSession(projectId, name, type, initialCommand?)` — returns sessionId string; sets as active if first for project
    - `closeSession(projectId, sessionId)` — removes from sessions array; if was active, select nearest neighbor (prefer left/earlier)
    - `switchSession(projectId, sessionId)` — sets activeSessionId[projectId]
    - `getProjectSessions(projectId)` — returns sessions[projectId] || []
    - `getActiveSession(projectId)` — returns the session matching activeSessionId[projectId]
    - `findAiSession(projectId)` — returns first session with type "ai" and status "running"
    - `markExited(projectId, sessionId)` — sets status to "exited"
    - `removeSession(projectId, sessionId)` — removes from array (used after exited delay)
    - `removeAllForProject(projectId)` — deletes sessions[projectId] and activeSessionId[projectId] (per D-11)
    - `getAllSessions()` — returns flat array of all sessions across all projects (for D-13 app quit)

    NOTE: PTY refs and xterm Terminal refs are NOT stored in the Zustand store. They will be managed by React refs in the component layer (Plan 02). The store only tracks session metadata. The graceful kill logic (SIGTERM + 3s + SIGKILL per D-12) will be a standalone exported async function `gracefulKillPty(pty: IPty): Promise<void>` in this same file, called by components that hold the PTY refs.

    Export the `gracefulKillPty` function:
    ```typescript
    export async function gracefulKillPty(pty: { kill: (signal?: string) => void; onExit: (cb: (exit: { exitCode: number }) => void) => void }): Promise<void> {
      try { pty.kill("SIGTERM"); } catch { return; }
      return new Promise<void>((resolve) => {
        const timeout = setTimeout(() => {
          try { pty.kill("SIGKILL"); } catch { /* already dead */ }
          resolve();
        }, 3000);
        pty.onExit(() => { clearTimeout(timeout); resolve(); });
      });
    }
    ```
