**src/types/agent.ts:**
```typescript
export type AgentStatus = "starting" | "running" | "idle" | "error" | "stopped";

export type AgentEntryType =
  | "execution_start"
  | "execution_complete"
  | "planning_complete"
  | "context_seeded"
  | "human_needed"
  | "error"
  | "approval_request";

export interface AgentActivityEntry {
  id: string;
  type: AgentEntryType;
  title: string;
  description: string;
  projectId?: string;
  projectName?: string;
  phaseId?: string;
  phaseName?: string;
  timestamp: number;
  approvalStatus?: "pending" | "approved" | "rejected";
}

export interface AgentState {
  // Panel
  panelOpen: boolean;
  activeTab: "activity" | "terminal";

  // Agent lifecycle
  status: AgentStatus;
  restartCount: number;

  // Activity
  entries: AgentActivityEntry[];

  // Computed
  pendingApprovalCount: () => number;

  // Actions
  togglePanel: () => void;
  setActiveTab: (tab: "activity" | "terminal") => void;
  setStatus: (status: AgentStatus) => void;
  addEntry: (entry: Omit<AgentActivityEntry, "id" | "timestamp">) => void;
  approveEntry: (entryId: string) => void;
  rejectEntry: (entryId: string) => void;
  clearEntries: () => void;
  incrementRestart: () => void;
  resetRestartCount: () => void;
}
```

**src/stores/useAgentStore.ts:**
- Import `create` from `zustand` (no persist middleware -- session-only state)
- Import `AgentState` from `@/types/agent`
- Create store with:
  - `panelOpen: false`
  - `activeTab: "activity"` (default per UI-SPEC)
  - `status: "starting"` (agent starts on launch per D-01)
  - `restartCount: 0`
  - `entries: []`
  - `pendingApprovalCount`: computed function filtering entries where `type === "approval_request" && approvalStatus === "pending"`
  - `togglePanel`: flip `panelOpen`
  - `setActiveTab`: set `activeTab`
  - `setStatus`: set `status`
  - `addEntry`: prepend new entry with `id: crypto.randomUUID()` and `timestamp: Date.now()` to entries array (newest first for reverse-chronological display)
  - `approveEntry(entryId)`: find entry by id, if found and approvalStatus is "pending", set to "approved"
  - `rejectEntry(entryId)`: find entry by id, if found and approvalStatus is "pending", set to "rejected"
  - `clearEntries`: set entries to []
  - `incrementRestart`: increment restartCount by 1
  - `resetRestartCount`: set restartCount to 0
