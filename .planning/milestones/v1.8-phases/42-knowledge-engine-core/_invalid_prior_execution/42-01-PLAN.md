```rust
pub fn get_skill_handler(&self, prefixed_name: &str) -> Result<(Arc<dyn SkillHandler>, String, Arc<SkillContext>), String> {
    let skill_reg = self.skill_registry.read().map_err(|e| e.to_string())?;
    let _skill = skill_reg.get(prefixed_name)
        .ok_or_else(|| format!("Skill '{}' not found. The plugin may need to be re-enabled.", prefixed_name))?;

    // Extract plugin name from prefixed name (e.g., "knowledge:ingest" -> "knowledge")
    let plugin_name = prefixed_name.split(':').next()
        .ok_or_else(|| format!("Invalid skill name format: {}", prefixed_name))?;
    let skill_name = prefixed_name.split(':').nth(1)
        .ok_or_else(|| format!("Invalid skill name format: {}", prefixed_name))?;

    let handler = self.skill_handlers.get(plugin_name)
        .ok_or_else(|| format!("No handler registered for plugin '{}'. The plugin may not support skill execution.", plugin_name))?
        .clone();

    let ctx = self.skill_context.as_ref()
        .ok_or_else(|| "Skill context not initialized".to_string())?
        .clone();

    Ok((handler, skill_name.to_string(), ctx))
}
```

**Create `src-tauri/src/plugins/core/knowledge.rs`:**

This file implements `KnowledgeSkillHandler` that wraps the existing `KnowledgeEngine`.

```rust
use std::sync::Arc;
use crate::knowledge::KnowledgeEngine;
use crate::knowledge::types::{SourceInput, SourceType};
use crate::plugins::{SkillHandler, SkillContext};

pub struct KnowledgeSkillHandler {
    engine: Arc<KnowledgeEngine>,
}

impl KnowledgeSkillHandler {
    pub fn new(engine: Arc<KnowledgeEngine>) -> Self {
        Self { engine }
    }
}

#[async_trait::async_trait]
impl SkillHandler for KnowledgeSkillHandler {
    async fn execute(
        &self,
        skill_name: &str,
        input: serde_json::Value,
        ctx: &SkillContext,
    ) -> Result<serde_json::Value, String> {
        // Get AI provider from context
        let provider = {
            let db = ctx.db.lock().map_err(|e| e.to_string())?;
            ctx.gateway.get_default_provider(&db).map_err(|e| e.to_string())?
        };

        match skill_name {
            "ingest" => {
                // Input: { "source_path": "string" } OR { "name": "string", "content": "string" }
                if let Some(source_path) = input.get("source_path").and_then(|v| v.as_str()) {
                    // File or URL path ingest
                    let (name, content, source_type) = if source_path.starts_with("http://") || source_path.starts_with("https://") {
                        (source_path.to_string(), String::new(), SourceType::Url)
                    } else {
                        let content = tokio::fs::read_to_string(source_path)
                            .await
                            .map_err(|e| format!("Failed to read source file '{}': {}", source_path, e))?;
                        let name = std::path::PathBuf::from(source_path)
                            .file_name()
                            .unwrap_or_default()
                            .to_string_lossy()
                            .to_string();
                        (name, content, SourceType::File)
                    };
                    let source = SourceInput { name, content, source_type };
                    let result = self.engine.ingest(source, provider.as_ref()).await.map_err(|e| e.to_string())?;
                    serde_json::to_value(result).map_err(|e| e.to_string())
                } else if let (Some(name), Some(content)) = (
                    input.get("name").and_then(|v| v.as_str()),
                    input.get("content").and_then(|v| v.as_str()),
                ) {
                    // Direct text ingest
                    let source = SourceInput {
                        name: name.to_string(),
                        content: content.to_string(),
                        source_type: SourceType::File,
                    };
                    let result = self.engine.ingest(source, provider.as_ref()).await.map_err(|e| e.to_string())?;
                    serde_json::to_value(result).map_err(|e| e.to_string())
                } else {
                    Err("Ingest requires either 'source_path' or both 'name' and 'content' fields".to_string())
                }
            }
            "query" => {
                let question = input.get("question")
                    .and_then(|v| v.as_str())
                    .ok_or_else(|| "Query requires a 'question' field".to_string())?;
                let result = self.engine.query(question, provider.as_ref()).await.map_err(|e| e.to_string())?;
                serde_json::to_value(result).map_err(|e| e.to_string())
            }
            "lint" => {
                let result = self.engine.lint(provider.as_ref()).await.map_err(|e| e.to_string())?;
                serde_json::to_value(result).map_err(|e| e.to_string())
            }
            _ => Err(format!("Unknown knowledge skill: {}", skill_name)),
        }
    }
}
```
