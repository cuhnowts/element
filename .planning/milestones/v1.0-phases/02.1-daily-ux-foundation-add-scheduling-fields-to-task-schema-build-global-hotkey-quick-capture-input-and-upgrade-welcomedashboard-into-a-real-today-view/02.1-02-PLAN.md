1. Create `src/components/center/TodayView.tsx`:

Define time grouping types and logic:
```typescript
type TimeGroup = "overdue" | "morning" | "afternoon" | "evening" | "unscheduled";
const TIME_GROUP_ORDER: TimeGroup[] = ["overdue", "morning", "afternoon", "evening", "unscheduled"];

function getTimeGroup(task: Task, today: string): TimeGroup {
  // Overdue: has due date before today AND not complete
  if (task.dueDate && task.dueDate < today && task.status !== "complete") {
    return "overdue";
  }
  // If task has scheduled time, use time-of-day buckets
  if (task.scheduledTime) {
    const hour = parseInt(task.scheduledTime.split(":")[0], 10);
    if (hour < 12) return "morning";
    if (hour < 17) return "afternoon";
    return "evening";
  }
  return "unscheduled";
}
```
