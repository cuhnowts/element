In `handleOpenAi`:
```typescript
const handleOpenAi = async () => {
  if (!directoryPath) {
    toast.error("Link a project directory first. The AI tool needs a directory to work in.");
    return;
  }

  // Phase 14: Check if tier dialog needed (per D-02)
  const needsTierDialog = !planningTier && !hasContent;
  if (needsTierDialog) {
    onTierDialogOpen();
    return;
  }

  // Existing flow continues for projects with a tier
  setIsLaunching(true);
  try {
    const contextPath = await api.generateContextFile(projectId);
    await api.startPlanWatcher(directoryPath);
    launchTerminalCommand("claude", ["--dangerously-skip-permissions", contextPath]);
  } catch (e) {
    toast.error(`Failed to launch AI: ${e}`);
  } finally {
    setIsLaunching(false);
  }
};
```
