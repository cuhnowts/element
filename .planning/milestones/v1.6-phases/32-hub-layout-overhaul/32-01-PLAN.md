Create `src/components/hub/HubToolbar.tsx` per D-01 and D-02:

```typescript
import { CalendarDays, Target } from "lucide-react";
import { Button } from "@/components/ui/button";
import { useWorkspaceStore } from "@/stores/useWorkspaceStore";

export function HubToolbar() {
  const calendarOpen = useWorkspaceStore((s) => s.hubCalendarOpen);
  const goalsOpen = useWorkspaceStore((s) => s.hubGoalsOpen);
  const toggleCalendar = useWorkspaceStore((s) => s.toggleHubCalendar);
  const toggleGoals = useWorkspaceStore((s) => s.toggleHubGoals);

  return (
    <div className="flex items-center gap-2 px-4 py-2 border-b border-border shrink-0">
      <Button
        variant={calendarOpen ? "default" : "ghost"}
        size="sm"
        onClick={toggleCalendar}
      >
        <CalendarDays className="mr-1.5 h-4 w-4" />
        Calendar
      </Button>
      <Button
        variant={goalsOpen ? "default" : "ghost"}
        size="sm"
        onClick={toggleGoals}
      >
        <Target className="mr-1.5 h-4 w-4" />
        Goals
      </Button>
    </div>
  );
}
```

Key points per decisions:
- Per D-01: Only Calendar and Goals buttons. No Briefing button.
- Per D-02: `variant="default"` when active (filled/accent), `variant="ghost"` when inactive.
- Per D-04: Calendar panel is left, Goals panel is right. Button order matches spatial position (Calendar first = left).
- Per D-05: Both can be active simultaneously -- no mutual exclusion logic.
- Use individual Zustand selectors per Pitfall 3: `(s) => s.hubCalendarOpen`, NOT `(s) => ({ calendarOpen: s.hubCalendarOpen, goalsOpen: s.hubGoalsOpen })`.
