**JumpToTop.tsx** -- create `src/components/hub/JumpToTop.tsx`:

Uses IntersectionObserver on a sentinel element (per RESEARCH.md Don't Hand-Roll table):
```typescript
import { useState, useEffect, useRef } from "react";
import { ChevronUp } from "lucide-react";
import { Button } from "@/components/ui/button";

interface JumpToTopProps {
  scrollRef: React.RefObject<HTMLDivElement | null>;
  sentinelRef: React.RefObject<HTMLDivElement | null>;
}

export function JumpToTop({ scrollRef, sentinelRef }: JumpToTopProps) {
  const [show, setShow] = useState(false);

  useEffect(() => {
    const sentinel = sentinelRef.current;
    if (!sentinel) return;
    const observer = new IntersectionObserver(
      ([entry]) => setShow(!entry.isIntersecting),
      { threshold: 0 }
    );
    observer.observe(sentinel);
    return () => observer.disconnect();
  }, [sentinelRef]);

  if (!show) return null;

  return (
    <Button
      variant="secondary"
      size="icon"
      className="fixed bottom-20 right-6 z-40 rounded-full shadow-lg"
      onClick={() => scrollRef.current?.scrollTo({ top: 0, behavior: "smooth" })}
      aria-label="Back to top"
    >
      <ChevronUp className="h-4 w-4" />
    </Button>
  );
}
```
- Per UI-SPEC: variant="secondary", size="icon", rounded-full, ChevronUp icon (16px), aria-label="Back to top"
- Position: fixed bottom-right, 16px from edges per UI-SPEC
- Visibility: shown only when sentinel (placed near top of scroll area, below action buttons) is not intersecting
- The sentinel `<div>` is placed in CommandHub, ref passed to JumpToTop

**Complete rewrite of `src/components/center/HubView.tsx`:**

Remove ALL existing code. The new HubView has no ResizablePanelGroup, no MinimizedColumn, no ColumnRibbon, no panel resize callbacks. Replace with:

```typescript
import { useWorkspaceStore } from "@/stores/useWorkspaceStore";
import { HubToolbar } from "@/components/hub/HubToolbar";
import { CommandHub } from "@/components/hub/CommandHub";
import { SlideOverPanel } from "@/components/hub/SlideOverPanel";
import { GoalsTreePanel } from "@/components/hub/GoalsTreePanel";
import { HubCalendar } from "@/components/hub/calendar/HubCalendar";

export function HubView() {
  const calendarOpen = useWorkspaceStore((s) => s.hubCalendarOpen);
  const goalsOpen = useWorkspaceStore((s) => s.hubGoalsOpen);

  return (
    <div className="relative h-full flex flex-col overflow-hidden">
      {/* Toolbar -- per D-01, sits at top of HubView */}
      <HubToolbar />

      {/* Center content -- always full width per D-03 */}
      <div className="relative flex-1 min-h-0">
        <CommandHub />

        {/* Overlay panels -- absolute positioned, float over center per D-03 */}
        <SlideOverPanel open={calendarOpen} side="left">
          <HubCalendar />
        </SlideOverPanel>
        <SlideOverPanel open={goalsOpen} side="right">
          <GoalsTreePanel />
        </SlideOverPanel>
      </div>
    </div>
  );
}
```

Key structural decisions:
- Outer div: `relative h-full flex flex-col overflow-hidden` -- relative for panel positioning, overflow-hidden to clip offscreen panels (Pitfall 6), flex-col for toolbar + content stack
- Toolbar: first child, `shrink-0` (already in HubToolbar)
- Content area: `relative flex-1 min-h-0` -- relative is the positioning context for absolute SlideOverPanels, flex-1 takes remaining height, min-h-0 prevents flex overflow
- SlideOverPanels: inside the content area div (not siblings of toolbar) so they don't cover the toolbar. They are `absolute top-0 bottom-0` within this relative container.
- Per D-03: Center view (CommandHub) stays full-width underneath panels. No flexbox or grid giving panels space.
- Per D-04: Calendar `side="left"`, Goals `side="right"`
- Per D-05: Both panels can be open simultaneously
- Per Pitfall 4: Both panels always rendered (not conditionally mounted). CSS transforms handle visibility.
- Zustand selectors: individual primitives per Pitfall 3

Removed imports: `react-resizable-panels` (ResizablePanelGroup, ResizablePanel, ResizableHandle, usePanelRef, PanelSize), `MinimizedColumn`, `ColumnRibbon`, `HubCenterPanel`. These are no longer used by HubView.
